def coprimes
  [].tap do |coprime_set|
    1.upto(self) do |i|
      coprime_set << i if i.coprime?(self)
    end
  end
end

def nsmooth(limit=2)
  ([0] * limit).tap do |ns|
    primes = Prime.each(self).to_a
    ns[0] = 1
    nextp = primes[0..primes.index(self)]

    indices = [0] * nextp.size
    (1...limit).each do |m|
      ns[m] = nextp.min
      (0...indices.size).each do |i|
        if ns[m] == nextp[i]
          indices[i] += 1
          nextp[i] = primes[i] * ns[indices[i]]
        end
      end
    end
  end
end
