def normalize
  l = denominators.lcm
  map{|r| Tonal::Ratio.new(l / r.denominator * r.numerator, l)}
end

def rpad!(min_size, value = nil)
  self.length > min_size ? self : (min_size - self.length).times { self << value }
  self
end
